// simplified example of React.lazy implementation
function lazy(dynamicImport) {
  let promise = null;
  let component = null;

  return () => {
    if (!promise) {
      promise = dynamicImport().then(module => {
        component = module.default;
      });
    }

    if (!component) {
      throw promise;
    }

    return component;
  };
}
